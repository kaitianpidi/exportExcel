        try {

                SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
                Date startDate = sdf.parse(checkStartDate);
                Date endDate = sdf.parse(checkEndDate);

                begin.setTime(startDate);
                Calendar end = Calendar.getInstance();
